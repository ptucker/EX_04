#include <iostream>
#include <ctime>
using namespace std;

Class Course
{
	private:
	
	string courseName;
	string student;
	int numberOfStudents;
	int capacity;
	
	public:
	
	Course()
	{
		courseName = 'Computer Science';
		student = 'George Whitworth'
		numberOfStudents = 1;
		capacity = 10;
	}
	
	int* addStudent(const int* numberOfStudents, int size)
	{
		size++;
		int* newarray = new int[size];
	
		for (int i = 0, count = 1; i < size; i++; count++)
		{
			newarray[i] = count;
		}
	
		return newarray;
	}

	void dropStudent(const int* numberOfStudents, int size)
	{
		size--;
		int* newarray = new int[size];
	
		for (int i = 0, count = 1; i < size; i++; count++)
		{
			newarray[i] = count;
		}
	}

	void Clear()
	{
		size = 0;
		int* newarray = new int[size];
	}

	string getCourseName()
	{
		return courseName;
	}	
	
	string getStudent()
	{
		return student;
	}
	
	const int getNumberOfStudents()
	{
		return numberOfStudents;
	}
}

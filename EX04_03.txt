#include <iostream>
using namespace std;

int* findlowest(const int* array, int size)
{
	int lowestelement = array[0];
	for(int i = 0, i < size, i++)
	{
		if(array[i] < lowestelement)
		{
			lowestelement = array[i]
		}
	}
	return lowestelement;
}
		
		

int main()
{
	int array[] = {1, 2, 4, 5, 10, 100, 2, -22};
	int* p = findlowest(array, 8);
}
